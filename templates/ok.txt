    <script>
        const form = document.getElementById('analysisForm');
        const fileInput = document.getElementById('file');
        const fileName = document.getElementById('fileName');
        const resultsSection = document.getElementById('resultsSection');
        const loading = document.getElementById('loading');
        const resultsContent = document.getElementById('resultsContent');

        // File input handler
        fileInput.addEventListener('change', function(e) {
            if (this.files.length > 0) {
                fileName.textContent = '‚úì ' + this.files[0].name + ' selected';
            }
        });

        // Drag and drop
        const fileLabel = document.querySelector('.file-input-label');
        fileLabel.addEventListener('dragover', (e) => {
            e.preventDefault();
            fileLabel.style.borderColor = 'rgba(251, 191, 36, 0.8)';
            fileLabel.style.background = 'rgba(251, 191, 36, 0.15)';
        });

        fileLabel.addEventListener('dragleave', () => {
            fileLabel.style.borderColor = 'rgba(251, 191, 36, 0.3)';
            fileLabel.style.background = 'rgba(251, 191, 36, 0.05)';
        });

        fileLabel.addEventListener('drop', (e) => {
            e.preventDefault();
            fileLabel.style.borderColor = 'rgba(251, 191, 36, 0.3)';
            fileLabel.style.background = 'rgba(251, 191, 36, 0.05)';
            fileInput.files = e.dataTransfer.files;
            fileName.textContent = '‚úì ' + e.dataTransfer.files[0].name + ' selected';
        });

        // Form submission
        form.addEventListener('submit', async function(e) {
            e.preventDefault();

            const formData = new FormData(this);
            
            // Show results section and loading state
            resultsSection.classList.add('active');
            loading.classList.add('active');
            resultsContent.style.display = 'none';

            try {
                // Send to backend
                const response = await fetch('/upload', {
                    method: 'POST',
                    body: formData
                });

                if (response.ok) {
                    const data = await response.json();
                    
                    // Simulate processing delay for better UX
                    setTimeout(() => {
                        // Update results
                        document.getElementById('resultImage').src = data.result_image;
                        document.getElementById('areaValue').textContent = Math.round(data.area);
                        document.getElementById('potentialValue').textContent = Math.round(data.solar_potential).toLocaleString();
                        document.getElementById('panelsValue').textContent = Math.round(data.num_panels);
                        document.getElementById('savingsValue').textContent = '‚Çπ' + Math.round(data.annual_savings).toLocaleString();
                        document.getElementById('savingsHighlight').textContent = '‚Çπ' + Math.round(data.annual_savings).toLocaleString();
                        
                        const payback = Math.round((data.solar_potential * 0.00005) / (data.annual_savings / 365));
                        document.getElementById('paybackPeriod').textContent = payback + ' years';

                        // Hide loading, show results
                        loading.classList.remove('active');
                        resultsContent.style.display = 'block';
                    }, 1500);
                } else {
                    alert('Error analyzing rooftop. Please try again.');
                    loading.classList.remove('active');
                }
            } catch (error) {
                console.error('Error:', error);
                alert('Error connecting to server. Please try again.');
                loading.classList.remove('active');
            }
        });
    </script>
